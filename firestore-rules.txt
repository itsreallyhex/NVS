rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {

    // تأكيد البريد (التحقق بخطوتين): البريد متأكَّد منه ولا لا.
    // request.auth.token.email_verified جايّة جوّه توكن موقّع من فايربيس —
    // ما يقدر أحد يزوّرها من أدوات المطوّر (Dev Tools). هذا هو الفرض الحقيقي للـ 2FA:
    // حتى لو شال المستخدم شاشة التأكيد من الواجهة، الخادم يرفض أي كتابة بدون بريد متأكَّد.
    function isVerified() {
      return request.auth != null
        && request.auth.token.email_verified == true;
    }

    // المسؤولون = الحسابات أصحاب هذه الـ UIDs فقط — وبشرط تأكيد البريد.
    // حتى الأدمن لا يتخطّى التحقق بخطوتين: لو بريده مو متأكَّد، يصير مثل أي زائر.
    // انسخ الـ UID من: Firebase Console → Authentication → Users
    function isAdmin() {
      return request.auth != null
        && isVerified()
        && request.auth.uid in [
          "kwbLZoAcamgD20RtZYUdt5fbmyo1", // Hex (itsreallyhex)
          "0zbVZGDbgWdTrfh42oJ7UBC24X23" // n_22j
        ];
    }

    // حساب الـ Worker (تأكيد الدفع فقط) — مفصول عن صلاحيات الأدمن.
    // لا يملك صلاحيات إدارية؛ يقرأ الطلب ليتحقق من المبلغ ويعدّل حقول الدفع فقط.
    function isWorker() {
      return request.auth != null
        && request.auth.uid == "5u4NpNAAvDM1YWPj4zm2FlNAcP43";
    }

    // صلاحية مسؤول يتحكّم فيها المالك من إعداداته. الافتراضي = مسموح (true) —
    // لو الحقل مو موجود أو وثيقة الإعدادات مفقودة ما نمنع شي (نحافظ على السلوك القديم).
    // المالك يتخطّى كل هذا (له فرع مستقل في كل قاعدة). كل تقييم يكلّف قراءة وحدة للإعدادات.
    function adminCan(perm) {
      return !exists(/databases/$(database)/documents/settings/store)
        || get(/databases/$(database)/documents/settings/store).data.get(perm, true) == true;
    }

    // الأقسام: أي شخص يقرأ — المسؤول يعدّل (إذا سمح له المالك)، المالك دائماً
    match /cats/{id} {
      allow read: if true;
      allow create, update: if isAdmin() && (isOwner() || adminCan("adminEditCatalog"));
      allow delete: if isAdmin() && (isOwner() || adminCan("adminDeleteCatalog"));
    }

    // المنتجات: أي شخص يقرأ — المسؤول يعدّل (إذا سمح له المالك)، المالك دائماً
    match /products/{id} {
      allow read: if true;
      allow create, update: if isAdmin() && (isOwner() || adminCan("adminEditCatalog"));
      allow delete: if isAdmin() && (isOwner() || adminCan("adminDeleteCatalog"));
    }

    // المالك = حساب Hex فقط — وبشرط تأكيد البريد كمان.
    // المالك نفسه ما يتخطّى الـ 2FA: بدون بريد متأكَّد ما يقدر يكتب أي شي إداري.
    function isOwner() {
      return request.auth != null
        && isVerified()
        && request.auth.uid == "kwbLZoAcamgD20RtZYUdt5fbmyo1";
    }

    // هل استقبال الطلبات موقوف الآن؟ نقرأها من إعدادات المتجر — يفرضها الخادم
    // فما يقدر أحد يتخطّاها من أدوات المطوّر. لو الوثيقة مو موجودة نعتبره غير موقوف
    // (آمن: ما نمنع الطلبات بالغلط). كل محاولة إنشاء طلب تكلّف قراءة وحدة لهذي الوثيقة.
    function ordersBlockedNow() {
      return exists(/databases/$(database)/documents/settings/store)
        && get(/databases/$(database)/documents/settings/store).data.get("ordersBlocked", false) == true;
    }

    // إعدادات المتجر (وضع الصيانة وغيرها):
    // أي شخص يقرأ — المالك فقط يعدّل
    match /settings/{id} {
      allow read: if true;
      allow write: if isOwner();
    }

    // أكواد الخصم: المالك وحده ينشئها/يعدّلها/يحذفها — الحارس الحقيقي هنا في القواعد،
    // فما يقدر أحد يضيف كوداً أو يغيّر نسبته من أدوات المطوّر.
    // - get: أي أحد يقرأ كوداً واحداً بالضبط (id = نص الكود) عشان يتحقق منه عند الدفع.
    // - list: المالك فقط — ما نخلي أحد يسرد كل الأكواد دفعة وحدة.
    // - الإنشاء/التعديل: المالك فقط، والقيم محدودة: نسبة 1..90 وفعّال/معطّل بوول.
    match /promocodes/{code} {
      allow get: if true;
      allow list: if isOwner();

      // المالك ينشئ/يعدّل بكل الحقول المسموحة بقيم محدودة:
      // percent 1..90، active بوول، maxUses عدد ≥0 (0 = بلا حد)، uses عدد ≥0،
      // expiresAt إمّا فاضي (بلا انتهاء) أو تاريخ.
      allow create, update: if isOwner()
        && request.resource.data.keys().hasOnly(["percent", "active", "createdAt", "maxUses", "uses", "expiresAt"])
        && request.resource.data.percent is number
        && request.resource.data.percent >= 1
        && request.resource.data.percent <= 90
        && request.resource.data.active is bool
        && (!("maxUses" in request.resource.data) || (request.resource.data.maxUses is number && request.resource.data.maxUses >= 0))
        && (!("uses" in request.resource.data) || (request.resource.data.uses is number && request.resource.data.uses >= 0))
        && (!("expiresAt" in request.resource.data) || request.resource.data.expiresAt == null || request.resource.data.expiresAt is timestamp);

      // العميل (مسجّل ومتحقّق) يزيد عدّاد الاستخدام +1 فقط — لا يلمس أي حقل ثاني،
      // وبشرط أن الكود فعّال وضمن الحد الأقصى وغير منتهي الصلاحية. هذا يجعل حدّ
      // الاستخدام فعّالاً فعلاً: عند الوصول للحد ما يقدر أحد يزيد العدّاد فيُرفض الطلب معه.
      allow update: if request.auth != null && isVerified()
        && request.resource.data.diff(resource.data).affectedKeys().hasOnly(["uses"])
        && request.resource.data.uses is number
        && request.resource.data.uses == resource.data.get("uses", 0) + 1
        && resource.data.active == true
        && (resource.data.get("maxUses", 0) == 0 || resource.data.get("uses", 0) < resource.data.maxUses)
        && (resource.data.get("expiresAt", null) == null || resource.data.expiresAt > request.time);

      allow delete: if isOwner();
    }

    // ملفات المستخدمين (اسم المستخدم + يوزر الديسكورد):
    // كل مستخدم يقرأ ويعدّل ملفه فقط — والمسؤول يقرأ الجميع
    match /users/{uid} {
      allow read: if request.auth != null
        && (request.auth.uid == uid || isAdmin());

      // الإنشاء (وقت التسجيل): الحقول الأربعة المعروفة فقط، البريد يطابق هوية الحساب،
      // وحد على طول الاسم/اليوزر. يمنع حقن حقول غريبة من البداية.
      allow create: if request.auth != null
        && request.auth.uid == uid
        && request.resource.data.keys().hasOnly(["username", "discord", "email", "createdAt"])
        && request.resource.data.email == request.auth.token.email
        && request.resource.data.username is string && request.resource.data.username.size() <= 40
        && request.resource.data.discord is string && request.resource.data.discord.size() <= 40;

      // التعديل: الاسم ويوزر الديسكورد فقط — ما يقدر يضيف حقول زي isAdmin/role،
      // ولا يكتب نصوص ضخمة. (التحديث في المتجر يدمج discord فقط.)
      allow update: if request.auth != null
        && request.auth.uid == uid
        && request.resource.data.diff(resource.data).affectedKeys().hasOnly(["username", "discord"])
        && request.resource.data.username is string && request.resource.data.username.size() <= 40
        && request.resource.data.discord is string && request.resource.data.discord.size() <= 40;
    }

    // الطلبات:
    // - إنشاء: المستخدم المسجّل لنفسه فقط
    // - قراءة: صاحب الطلب أو المسؤول
    // - تعديل: المسؤول — أو صاحب الطلب لإلغائه فقط وقبل بدء التنفيذ
    // - حذف: المسؤول فقط
    match /orders/{id} {
      allow create: if request.auth != null
        // التحقق بخطوتين إجباري: ما أحد ينشئ طلب وبريده مو متأكَّد — حتى من Dev Tools.
        && isVerified()
        && request.resource.data.uid == request.auth.uid
        && request.resource.data.email == request.auth.token.email
        && request.resource.data.total is number
        && request.resource.data.total >= 1
        && request.resource.data.items is list
        && request.resource.data.items.size() <= 50
        && request.resource.data.status == "قيد المراجعة"
        // لا يجوز للعميل إنشاء طلب مُعلَّم كمدفوع — الدفع يؤكَّد من الخادم/الإدارة فقط
        && request.resource.data.get('paid', false) == false
        // إيقاف الطلبات: لو المالك موقف الاستقبال، ما أحد يقدر ينشئ طلب — حتى من أدوات
        // المطوّر. الإدارة تتخطى (تقدر تطلب للاختبار وقت الإيقاف).
        && (isAdmin() || !ordersBlockedNow())
        // كود الخصم: لو الطلب يدّعي كوداً، لازم يكون الكود موجوداً وفعّالاً ونسبته تطابق
        // المعلنة بالضبط، وغير منتهي الصلاحية، وما وصل حدّ الاستخدام — يفرضه الخادم.
        // فما يقدر أحد يخترع نسبة خصم أو يستخدم كوداً منتهياً/مستهلَكاً من أدوات المطوّر.
        && (!("promoCode" in request.resource.data)
          || (request.resource.data.promoCode is string
            && request.resource.data.promoPercent is number
            && exists(/databases/$(database)/documents/promocodes/$(request.resource.data.promoCode))
            && get(/databases/$(database)/documents/promocodes/$(request.resource.data.promoCode)).data.active == true
            && get(/databases/$(database)/documents/promocodes/$(request.resource.data.promoCode)).data.percent == request.resource.data.promoPercent
            && (get(/databases/$(database)/documents/promocodes/$(request.resource.data.promoCode)).data.get("expiresAt", null) == null
              || get(/databases/$(database)/documents/promocodes/$(request.resource.data.promoCode)).data.expiresAt > request.time)
            && (get(/databases/$(database)/documents/promocodes/$(request.resource.data.promoCode)).data.get("maxUses", 0) == 0
              || get(/databases/$(database)/documents/promocodes/$(request.resource.data.promoCode)).data.get("uses", 0) < get(/databases/$(database)/documents/promocodes/$(request.resource.data.promoCode)).data.maxUses)));
      allow read: if request.auth != null
        && (resource.data.uid == request.auth.uid || isAdmin() || isWorker());
      // التعديل:
      // - المالك: صلاحية كاملة دائماً — يغيّر الحالة حتى بعد الإكمال، ويعدّل التكليفات بلا قيد.
      // - المسؤول العادي: يعدّل الطلبات ضمن قيدين:
      //     (أ) قفل الإكمال: لا يغيّر حالة طلب "مكتمل".
      //     (ب) فصل المهام: لا يعدّل التكليفات إذا كان الطلب مكتملاً أو كان هو نفسه أحد المكلّفين.
      // - صاحب الطلب: إلغاء طلبه غير المدفوع قبل بدء التنفيذ فقط.
      // العميل لا يستطيع تعليم الطلب كمدفوع إطلاقاً؛ حالة "paid" تُكتب من الإدارة
      // أو من دالة خادمية موثوقة بعد التحقق من ميسر فقط.
      allow update: if isOwner()
        // الـ Worker: تأكيد الدفع فقط — لا يلمس إلا حقول الدفع، ولا شيء غيرها.
        || (isWorker()
          && request.resource.data.diff(resource.data).affectedKeys()
              .hasOnly(["paid", "paidAt", "moyasarPaymentId"]))
        || (isAdmin()
          && !(resource.data.status == "مكتمل"
            && request.resource.data.diff(resource.data).affectedKeys().hasAny(["status"]))
          && !(request.resource.data.diff(resource.data).affectedKeys().hasAny(["assignees", "assigneeNames"])
            && (resource.data.status == "مكتمل"
              || request.auth.uid in resource.data.get("assignees", [])))
          // صلاحيات يتحكّم فيها المالك: كل نوع تعديل يتطلّب صلاحيته (المالك يتخطّى أصلاً بفرعه فوق)
          && (!request.resource.data.diff(resource.data).affectedKeys().hasAny(["status"])
            || adminCan("adminChangeStatus"))
          && (!request.resource.data.diff(resource.data).affectedKeys().hasAny(["paid", "paidAt", "moyasarPaymentId"])
            || adminCan("adminMarkPaid"))
          && (!request.resource.data.diff(resource.data).affectedKeys().hasAny(["assignees", "assigneeNames"])
            || adminCan("adminAssign")))
        || (request.auth != null
          && isVerified()
          && resource.data.uid == request.auth.uid
          && resource.data.status in ["قيد المراجعة", "قيد الانتظار"]
          && resource.data.paid != true
          && request.resource.data.status == "ملغي"
          && request.resource.data.diff(resource.data).affectedKeys()
              .hasOnly(["status", "cancelledBy"]));
      // الحذف: المالك دائماً، والمسؤول إذا سمح له المالك — أو الـ Worker للطلبات غير
      // المدفوعة "قيد المراجعة" فقط (التنظيف التلقائي بعد 4 أيام). لا يحذف الـ Worker
      // أي طلب مدفوع أو قيد التنفيذ أو مكتمل.
      allow delete: if (isAdmin() && (isOwner() || adminCan("adminDeleteOrders")))
        || (isWorker()
          && resource.data.paid != true
          && resource.data.status == "قيد المراجعة");

      // سجل التغييرات (تدقيق): إنشاء فقط من الإدارة — لا تعديل ولا حذف إطلاقاً.
      // سجلّ غير قابل للعبث: حتى الإدارة لا تستطيع تحرير أو حذف أي قيد بعد كتابته.
      match /statusLog/{logId} {
        allow read: if isAdmin();
        allow create: if isAdmin();
        allow update, delete: if false;
      }
    }

    // قراءة جماعية لكل سجلات الطلبات (collection group) — لسجل النشاط الكامل في لوحة الإدارة
    match /{path=**}/statusLog/{logId} {
      allow read: if isAdmin();
    }

    // سجل أحداث المتجر (إضافة/تعديل/حذف منتجات وأقسام): إنشاء وقراءة للإدارة، لا تعديل/حذف
    match /storeLog/{logId} {
      allow read: if isAdmin();
      allow create: if isAdmin();
      allow update, delete: if false;
    }
  }
}
